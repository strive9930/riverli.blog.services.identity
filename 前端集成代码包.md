### 2. src/store/modules/user.ts (替换整个文件)

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import service from '@/utils/request'
import { ElMessage } from 'element-plus'
import type { LoginResponseData, UserInfo } from '@/types/auth'

export const useUserStore = defineStore('user', () => {
  // 状态
  const accessToken = ref<string>(localStorage.getItem('access_token') || '')
  const userInfo = ref<UserInfo | null>(
    (() => {
      const stored = localStorage.getItem('user_info')
      if (!stored || stored === 'undefined') return null
      try {
        return JSON.parse(stored)
      } catch (e) {
        console.warn('用户信息JSON解析失败:', e)
        return null
      }
    })()
  )

  // 计算属性
  const isLoggedIn = computed(() => !!accessToken.value)
  const email = computed(() => userInfo.value?.email || '')
  const nickName = computed(() => userInfo.value?.nickName || userInfo.value?.nickname || '')
  const avatar = computed(() => userInfo.value?.avatar || '')
  const roles = computed(() => userInfo.value?.roles || [])
  const permissions = computed(() => userInfo.value?.permissions || [])
  const menus = computed(() => userInfo.value?.menus || [])
  const isAdmin = computed(() => userInfo.value?.isAdmin || false)
  const roleNames = computed(() => roles.value.map(r => r.name))
  const userId = computed(() => userInfo.value?.id || '')

  // Actions
  const login = async (credentials: { email: string; password: string }): Promise<{ success: boolean; message?: string }> => {
    console.log('开始登录流程，凭据:', credentials);
    try {
      const response = await service.post('/auth/login', credentials);
      console.log('收到登录响应:', response);
      
      // 检查根级别的success字段
      if (response.success && response.data) {
        const loginData: LoginResponseData = response.data;
        
        if (!loginData.token) {
          console.error('登录数据格式错误:', loginData);
          return { success: false, message: '登录数据格式错误' };
        }
        
        // 构建用户信息对象
        const newUserInfo: UserInfo = {
          id: loginData.userId,
          username: loginData.username,
          email: loginData.email,
          nickName: loginData.nickname,
          avatar: loginData.avatar,
          roles: loginData.roles,
          permissions: loginData.permissions,
          menus: loginData.menus,
          isAdmin: loginData.isAdmin,
          createdAt: loginData.createdAt
        };
        
        console.log('设置的用户信息:', newUserInfo);
        
        // 存储token和用户信息
        accessToken.value = loginData.token;
        userInfo.value = newUserInfo;
        
        // 持久化到localStorage
        localStorage.setItem('access_token', accessToken.value);
        localStorage.setItem('user_info', JSON.stringify(userInfo.value));
        
        console.log('登录成功，用户信息已保存');
        console.log('- Token:', accessToken.value.substring(0, 20) + '...');
        console.log('- 用户ID:', newUserInfo.id);
        console.log('- 用户名:', newUserInfo.username);
        console.log('- 角色数量:', newUserInfo.roles.length);
        console.log('- 权限数量:', newUserInfo.permissions.length);
        console.log('- 菜单数量:', newUserInfo.menus.length);
        console.log('- 是否管理员:', newUserInfo.isAdmin);
        
        return { success: true };
      } else {
        console.log('登录失败:', response.message);
        return { success: false, message: response.message || '登录失败' };
      }
    } catch (error: any) {
      console.error('登录失败:', error)
      return { 
        success: false, 
        message: error.response?.data?.message || '登录失败，请检查网络连接' 
      }
    }
  }

  const logout = () => {
    // 清除所有用户相关数据
    accessToken.value = ''
    userInfo.value = null
    
    // 清除localStorage
    localStorage.removeItem('access_token')
    localStorage.removeItem('user_info')
    localStorage.removeItem('routes')
    localStorage.removeItem('menus')
    localStorage.removeItem('menuTree')
    
    ElMessage.success('退出登录成功')
  }

  const initUserState = async () => {
    console.log('=== 开始初始化用户状态 ===')
    const storedToken = localStorage.getItem('access_token')
    const storedUserInfo = localStorage.getItem('user_info')
    
    console.log('初始化用户状态:')
    console.log('- Token存在:', !!storedToken)
    console.log('- 用户信息存在:', !!storedUserInfo)
    
    if (storedToken && storedUserInfo && storedUserInfo !== 'undefined') {
      accessToken.value = storedToken
      try {
        userInfo.value = JSON.parse(storedUserInfo)
        console.log('从localStorage恢复用户状态成功')
        console.log('- 用户ID:', userInfo.value?.id)
        console.log('- 用户名:', userInfo.value?.username)
        console.log('- 角色数量:', userInfo.value?.roles.length)
        console.log('- 权限数量:', userInfo.value?.permissions.length)
        console.log('- 菜单数量:', userInfo.value?.menus.length)
      } catch (e) {
        console.warn('初始化用户状态时JSON解析失败:', e)
        logout()
      }
    } else {
      console.log('没有找到存储的用户状态')
    }
    console.log('=== 用户状态初始化完成 ===')
  }

  return {
    // 状态
    accessToken,
    userInfo,
    
    // 计算属性
    isLoggedIn,
    email,
    nickName,
    avatar,
    roles,
    roleNames,
    permissions,
    menus,
    isAdmin,
    userId,
    
    // 方法
    login,
    logout,
    initUserState
  }
})
```

import type { MenuTreeItem } from '@/types/auth'

// 在 store 中添加这个方法
const generateRoutesFromMenus = async (menus: MenuTreeItem[]) => {
  console.log('开始从菜单生成路由，菜单数量:', menus.length)
  
  const dynamicRoutes: any[] = []
  
  menus.forEach(menu => {
    const route = convertMenuToRoute(menu)
    if (route) {
      dynamicRoutes.push(route)
    }
  })
  
  routes.value = dynamicRoutes
  isRoutesLoaded.value = true
  
  console.log('路由生成完成，路由数量:', dynamicRoutes.length)
  
  return dynamicRoutes
}

const convertMenuToRoute = (menu: MenuTreeItem): any | null => {
  if (!menu.path) return null
  
  const route: any = {
    path: menu.path,
    name: menu.name,
    meta: {
      title: menu.title,
      icon: menu.icon,
      permissions: menu.requiredPermission ? [menu.requiredPermission] : []
    }
  }
  
  // 如果有子菜单，递归处理
  if (menu.children && menu.children.length > 0) {
    route.children = menu.children
      .map(child => convertMenuToRoute(child))
      .filter((r): r is any => r !== null)
  }
  
  return route
}

// 导出新方法
return {
  // ... 其他返回值
  generateRoutesFromMenus,
  // ...
}

### 4. src/router/index.ts (更新路由守卫)

在路由守卫中使用新的登录数据：

```typescript
import { useUserStore } from '@/store/modules/user'
import { usePermissionStore } from '@/store/modules/permission'

router.beforeEach(async (to, from, next) => {
  const userStore = useUserStore()
  const permissionStore = usePermissionStore()
  
  if (userStore.isLoggedIn) {
    if (to.path === '/login') {
      next({ path: '/' })
    } else {
      // 如果还没有加载动态路由，则使用登录时获取的菜单数据生成路由
      if (!permissionStore.isRoutesLoaded && userStore.menus.length > 0) {
        try {
          console.log('使用登录时获取的菜单数据生成路由')
          await permissionStore.generateRoutesFromMenus(userStore.menus)
          next({ ...to, replace: true })
        } catch (error) {
          console.error('生成路由失败:', error)
          userStore.logout()
          next(`/login?redirect=${to.path}`)
        }
      } else {
        next()
      }
    }
  } else {
    if (to.path === '/login') {
      next()
    } else {
      next(`/login?redirect=${to.path}`)
    }
  }
})
```

## 🔄 可选：保留旧接口作为备用

如果您想保留旧的接口调用方式作为备用或调试用途，可以创建一个辅助服务：

```typescript
// src/services/authService.ts

import service from '@/utils/request'
import type { UserInfo } from '@/types/auth'

export class AuthService {
  /**
   * 刷新用户信息（可选，用于强制刷新）
   */
  static async refreshUserInfo(): Promise<UserInfo | null> {
    try {
      const response = await service.get('/user-frontend/info')
      if (response.success && response.data) {
        return response.data
      }
      return null
    } catch (error) {
      console.error('刷新用户信息失败:', error)
      return null
    }
  }

  /**
   * 刷新权限列表（可选）
   */
  static async refreshPermissions(): Promise<string[]> {
    try {
      const response = await service.get('/user-frontend/permissions')
      if (response.success && response.data) {
        return response.data
      }
      return []
    } catch (error) {
      console.error('刷新权限失败:', error)
      return []
    }
  }

  /**
   * 刷新菜单树（可选）
   */
  static async refreshMenus(): Promise<any[]> {
    try {
      const response = await service.get('/user-frontend/menus')
      if (response.success && response.data) {
        return response.data
      }
      return []
    } catch (error) {
      console.error('刷新菜单失败:', error)
      return []
    }
  }
}
```
